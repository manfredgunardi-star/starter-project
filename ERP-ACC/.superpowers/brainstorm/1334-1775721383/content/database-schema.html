Database Schema — Entity Relationship

Relasi antar tabel utama dalam aplikasi ERP

Master Data

products

id, sku, name, category
base_unit_id → units
buy_price, sell_price
is_taxable (PPN toggle), tax_rate
is_active, created_at

units

id, name (pcs, dus, kg, liter...)

unit_conversions

id, product_id → products
from_unit_id → units
to_unit_id → units
conversion_factor (misal: 1 dus = 12 pcs → factor = 12)

customers

id, name, address, phone, email
npwp, is_active, created_at

suppliers

id, name, address, phone, email
npwp, is_active, created_at

coa (Chart of Accounts)

id, code (1-10000), name
type (asset, liability, equity, revenue, expense)
normal_balance (debit/credit)
parent_id → coa (self-ref, hirarki)
is_active, created_at
Transaksi

sales_orders

id, so_number, date
customer_id → customers
status (draft, confirmed, invoiced, done)
subtotal, tax_amount, total

sales_order_items

id, sales_order_id → sales_orders
product_id → products
unit_id → units
quantity, unit_price, tax_amount, total

invoices

id, invoice_number, date, due_date
type (sales / purchase)
customer_id / supplier_id
sales_order_id → sales_orders (nullable)
subtotal, tax_amount, total
amount_paid, status (draft, posted, paid, partial)

purchase_orders

id, po_number, date
supplier_id → suppliers
status (draft, confirmed, received, done)
subtotal, tax_amount, total

purchase_order_items

id, purchase_order_id → purchase_orders
product_id → products
unit_id → units
quantity, unit_price, tax_amount, total

payments

id, payment_number, date
type (incoming / outgoing)
invoice_id → invoices
account_id → accounts
amount, notes
Inventory

inventory_movements

id, date, product_id → products
type (in / out / adjustment)
quantity_base (selalu dalam satuan dasar)
unit_id, quantity_original
reference_type (purchase, sale, adjustment)
reference_id, notes

inventory_stock

id, product_id → products
quantity_on_hand (satuan dasar)
last_updated
Kas/Bank & Accounting

accounts (Kas/Bank)

id, name (Kas Besar, BCA, Mandiri...)
type (cash / bank)
coa_id → coa
balance, is_active

journals

id, journal_number, date, description
source (auto / manual)
reference_type, reference_id
is_posted, created_by

journal_items

id, journal_id → journals
coa_id → coa
debit, credit, description

bank_reconciliations

id, account_id → accounts
date, statement_balance
system_balance, is_reconciled
Auth & System

profiles

id → auth.users (Supabase Auth)
full_name, role (admin, staff, viewer)
is_active, created_at

audit_logs

id, table_name, record_id
action (create, update, delete)
old_data, new_data (JSONB)
user_id → profiles, created_at
