rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() { return request.auth != null; }

    // Role lookup dari dokumen users/{uid}
    function myUserDoc() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }
    function myRole() {
      return signedIn() && myUserDoc().exists() ? myUserDoc().data.role : 'reader';
    }
    function inRoles(roles) {
      return signedIn() && myRole() in roles;
    }
    function isSuperAdmin() { return signedIn() && myRole() == 'superadmin'; }
    function isOwner() { return signedIn() && myRole() == 'owner'; }
    function isAdminSJ() { return signedIn() && myRole() == 'admin_sj'; }
    function isAdminInv() { return signedIn() && myRole() == 'admin_invoice'; }
    function isAdminKeu() { return signedIn() && myRole() == 'admin_keuangan'; }

    function isStaff() {
      return signedIn() && myRole() in ['superadmin','owner','admin_sj','admin_invoice','admin_keuangan'];
    }

    // Admin Invoice boleh update Surat Jalan TERBATAS untuk menandai ter-invoice (tanpa bisa ubah data operasional).
    function sjInvoiceFieldsOnly() {
      // Batasi field yang boleh berubah saat admin_invoice melakukan update
      return request.resource.data.diff(resource.data).changedKeys().hasOnly([
        'statusInvoice',
        'invoiceId',
        'invoiceNo',
        'updatedAt',
        'updatedBy'
      ]);
    }

    // --- USERS ---
    match /users/{uid} {
      // semua user yang login boleh baca daftar user? (dibutuhkan untuk admin panel)
      // demi keamanan: hanya admin/owner/superadmin yang boleh baca semua; user biasa hanya baca doc sendiri
      allow read: if signedIn() && (request.auth.uid == uid || inRoles(['superadmin','owner','admin_sj','admin_invoice','admin_keuangan']));

      // bootstrap doc sendiri saat pertama login
      allow create: if signedIn()
                    && request.auth.uid == uid
                    && request.resource.data.role == 'reader';

      // update profil sendiri tanpa mengubah role
      allow update: if signedIn()
                    && request.auth.uid == uid
                    && !( 'role' in request.resource.data );

      // superadmin boleh set role + manage user docs
      allow update, delete: if isSuperAdmin();
    }

    // --- SETTINGS ---
    match /settings/{docId} {
      allow read: if signedIn();
      allow write: if isSuperAdmin() || isOwner();
    }

    // --- MASTER DATA ---
    match /trucks/{id} {
      allow read: if signedIn();
      allow write: if isSuperAdmin() || isAdminSJ();
    }
    match /supir/{id} {
      allow read: if signedIn();
      allow write: if isSuperAdmin() || isAdminSJ();
    }
    match /rute/{id} {
      allow read: if signedIn();
      allow write: if isSuperAdmin() || isAdminSJ();
    }
    match /material/{id} {
      allow read: if signedIn();
      allow write: if isSuperAdmin() || isAdminSJ();
    }

    // --- TRANSAKSI ---
    match /surat_jalan/{id} {
      allow read: if signedIn();
      allow create: if isSuperAdmin() || isAdminSJ();
      allow update: if isSuperAdmin() || isAdminSJ() || (isAdminInv() && sjInvoiceFieldsOnly());
      allow delete: if isSuperAdmin(); // prefer soft delete
    }

    // Legacy collection name (camelCase) for backward compatibility
    match /suratJalan/{id} {
      allow read: if signedIn();
      // Keep write permissions aligned with surat_jalan
      allow create: if isSuperAdmin() || isAdminSJ();
      allow update: if isSuperAdmin() || isAdminSJ() || (isAdminInv() && sjInvoiceFieldsOnly());
      allow delete: if isSuperAdmin();
    }

    match /invoice/{id} {
      allow read: if signedIn();
      allow create, update: if isSuperAdmin() || isAdminInv();
      allow delete: if isSuperAdmin();
    }

    // Some older builds used plural collection name
    match /invoices/{id} {
      allow read: if signedIn();
      allow create, update: if isSuperAdmin() || isAdminInv();
      allow delete: if isSuperAdmin();
    }

    match /biaya/{id} {
      allow read: if signedIn() && inRoles(['superadmin','owner','admin_keuangan','admin_sj','admin_invoice','reader']);
      allow create, update: if isSuperAdmin() || isAdminKeu();
      allow delete: if isSuperAdmin();
    }

    match /transaksi/{id} {
      // Semua role internal boleh melihat laporan kas (UI filter tetap pakai isActive)
      allow read: if isStaff();

      // Superadmin & Keuangan full access
      allow create, update: if isSuperAdmin() || isAdminKeu();

      // Admin SJ boleh membuat transaksi Uang Jalan (pengeluaran) dan hanya boleh soft-delete/restore
      allow create: if isAdminSJ()
        && request.resource.data.keys().hasAll(['id','tipe','nominal','pt','tanggal','keterangan','createdAt','createdBy','isActive','suratJalanId'])
        && request.resource.data.tipe == 'pengeluaran'
        && request.resource.data.isActive == true
        && request.resource.data.nominal is int
        && request.resource.data.nominal >= 0
        && request.resource.data.suratJalanId is string;

      allow update: if isAdminSJ()
        && resource.data.suratJalanId is string
        // hanya boleh ubah status aktif & audit trail
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(['isActive','updatedAt','updatedBy']);

      allow delete: if isSuperAdmin();
    }

    // --- HISTORY LOG ---
    match /history_log/{id} {
      allow read: if signedIn() && inRoles(['superadmin','owner','admin_sj','admin_invoice','admin_keuangan']);
      allow create: if signedIn() && inRoles(['superadmin','admin_sj','admin_invoice','admin_keuangan']);
      allow update, delete: if false; // append-only
    }

    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
